#Country to Province

	title = "EVTNAME3677402"
	desc = "EVTDESC3677402"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
			has_global_flag = zombie_end
			any_owned_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

	mean_time_to_happen = {
		months = 2
	}

	option = {
		name = "EVTOPTA3677402"
		set_country_flag = zombie_outbreak
		any_pop = {
			consciousness = 1
			militancy = -2
		}
		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			any_pop = {
				reduce_pop = 0.8
				consciousness = -3
				militancy = 10
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
		}
	}
}

	title = "EVTNAME99025"

		province_control_days=1800 #5 years
		NOT = { owner = { tag = ZOM } }

		controlled_by = REB
		exists = ZOM
		ZOM = {
			has_country_modifier = accepts_new_provinces
		}
	}

		months = 24
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
		modifier = {
			factor = 0.75
			has_empty_adjacent_province = yes
		}
	}

	option = {
		name = "Uh oh."

			reduce_pop = 0.4
			militancy = 8
			consciousness = -3
		}
		add_core = THIS
		owner = { prestige = -3 }

		change_controller = ZOM
		secede_province = ZOM
		ZOM = {
			remove_country_modifier = do_not_annex
			add_country_modifier = {
				name = do_not_annex
				duration = 31
			}

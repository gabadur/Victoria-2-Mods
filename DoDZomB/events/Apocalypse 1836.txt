#Outbreak
country_event = {
	id = 999901
	title = "OUTBREAK_START_TITLE"
	desc = "OUTBREAK_START_DESC"
	picture = disease

	fire_only_once = yes
	
	trigger = {
		ai = no
		has_global_flag = zombie_start
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "OUTBREAK_START_OPT"
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				continent = asia
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				continent = europe
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
				OR = {
				continent = north_america
				continent = south_america
				}
			}
			country_event = 999902
		}
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 1
			}
			country_event = 999902
		}
	}
}

country_event = {
	id = 999902
	title = "OUTBREAK_START_LOCAL_TITLE"
	desc = "OUTBREAK_START_LOCAL_DESC"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
			any_owned_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

#Province Victory
province_event = {
	id = 999914
	title = "OUTBREAK_PLAYER_VICTORY_TITLE"
	desc = "OUTBREAK_PLAYER_VICTORY_DESC"

	trigger = {
		has_province_modifier = zombie_outbreak
		NOT = {
			controlled_by = REB
			country_units_in_province = REB
		}
		units_in_province = 10
	}

	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 1.25
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

	option = {
		name = "Excellent."
		remove_province_modifier = zombie_outbreak
		any_pop = {
			consciousness = 2
			militancy = -8
		}
	}
}

			}
		}

	}

	option = {
		name = "OUTBREAK_SPREAD_LOCAL_OPT"
			random_province = {
			limit = {
				has_province_modifier = zombie_outbreak
			}
			random_neighbor_province = {
			limit = {
				NOT = { has_province_modifier = zombie_outbreak }
				NOT = { units_in_province = 10 }
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
			}
			}
			random_province = {
			limit = {
				has_province_modifier = zombie_outbreak
			}
			any_neighbor_country = {
			limit = {
				neighbor = THIS
			}
			random_neighbor_province = {
			limit = {
				NOT = { has_province_modifier = zombie_outbreak }
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
			}
			}
			}
		}
}
